<script>
// ===== MENU ITEMS =====
const menuItems = [
"Cortado","Machiato","Cappuccino","Cafe Latte","Flat White","Mocha Latte",
"Espresso","Doppio","Americano","Romano","Lungo",
"Classic Coffee Frappe","Oreo Frappe","Kitkat Frappe","Java Chip Frappe","Hazlenut Frappe","Mocha Frappe",
"Iced Americano","Iced Latte","Vietnamese Iced","Iced Shaken Espresso","Cold Brew",
"RedXpress","PinkXpress","GingerXpress","CocoXpress","FrozenXpress","TonicXpress","CranXpress",
"Kimchi","Carbonara","Garlic Chilli Oil","Shin Chilli","Gochujang Paneer","Burnt Mushroom & Pepper","Corn & Cheese Ramen","Dragon's Breath",
"Mix Veg Momos","Spicy Veg Momos","Maggie Momos","Paneer Momos","Veg Cheese Momos","Paneer Tikka Momos",
"Sandwich","Burger","French Fries","Salad","Pizza Slice","Pasta"
];

// ===== AUTOCOMPLETE LOGIC =====
const searchInput = document.getElementById('searchInput');
const autocompleteList = document.getElementById('autocomplete-list');

searchInput.addEventListener('input', function(){
    const val = this.value.toLowerCase();
    autocompleteList.innerHTML = '';
    if(!val) return;
    menuItems.forEach(item => {
        if(item.toLowerCase().startsWith(val)){
            const div = document.createElement('div');
            div.classList.add('autocomplete-item');
            div.innerHTML = `<strong>${item.substr(0,val.length)}</strong>${item.substr(val.length)}`;
            div.addEventListener('click', function(){
                searchInput.value = item;
                autocompleteList.innerHTML = '';
                searchMenu(item);
            });
            autocompleteList.appendChild(div);
        }
    });
});

// ===== SEARCH FUNCTION =====
function searchMenu(item=null){
    const query = item || searchInput.value.trim();
    if(!query) return;
    window.location.href = `menu.html?search=${encodeURIComponent(query)}`;
}

// Close dropdown if clicked outside
document.addEventListener('click', function(e){
    if(e.target !== searchInput) autocompleteList.innerHTML='';
});
