<script>
// CAROUSEL
function initCarousel(id, prev, next){
    const c = document.getElementById(id);
    const cards = c.children;
    let angle = 0;
    const count = cards.length;
    const theta = 360 / count;
    const radius = 320;

    for(let i=0;i<count;i++){
        cards[i].style.transform = `rotateY(${i*theta}deg) translateZ(${radius}px)`;
    }

    function rotate(){
        c.style.transform = `translateX(-50%) rotateY(${angle}deg)`;
    }

    document.getElementById(prev).onclick = ()=>{ angle += theta; rotate(); };
    document.getElementById(next).onclick = ()=>{ angle -= theta; rotate(); };
}

initCarousel('drinksCarousel','prevDrinks','nextDrinks');
initCarousel('foodCarousel','prevFood','nextFood');

// CART FUNCTIONALITY
let cart = [];

const orderButtons = document.querySelectorAll('.order-btn');
orderButtons.forEach(btn=>{
    btn.addEventListener('click',()=>{
        const card = btn.closest('.carousel-card');
        const item = {
            name: card.dataset.name,
            price: card.dataset.price,
            image: card.dataset.image,
            quantity: 1
        };
        // Check if already in cart
        const existing = cart.find(i=>i.name===item.name);
        if(existing){ existing.quantity++; }
        else{ cart.push(item); }
        alert(`${item.name} added to cart!`);
        localStorage.setItem('cart', JSON.stringify(cart));
    });
});

// GO TO CHECKOUT
document.getElementById('goCheckout').addEventListener('click',()=>{
    localStorage.setItem('cart', JSON.stringify(cart));
    window.location.href='checkout.html';
});
</script>
